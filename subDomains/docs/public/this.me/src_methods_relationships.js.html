//src/methods/relationships.js
/**
 * @module relationships
 * @description Methods to manage relationships for the `Me` class.
 */

/**
 * Add a relationship to the identity.
 * @function
 * @param {Array} relationships - The relationships array to update.
 * @param {Object} relationship - The relationship to add (e.g., { username: "alice", status: "friend" }).
 * @throws {Error} If the relationship is invalid.
 */
export function addRelationship(relationships, relationship) {
    if (!relationship || typeof relationship !== 'object') {
      throw new Error('Invalid relationship');
    }
    relationships.push(relationship);
  }
  
  /**
   * Retrieve all relationships for the identity.
   * @function
   * @param {Array} relationships - The relationships array to retrieve from.
   * @returns {Array} The relationships array.
   * @instance
   */
  export function getRelationships(relationships) {
    return relationships;
  }
